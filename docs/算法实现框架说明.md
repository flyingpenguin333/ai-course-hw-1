```python
def cannon_moves(point, player, board):
    moves = []
    for dr, dc in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
        r, c = point.row + dr, point.col + dc
        # 阶段1：移动（不吃子），遇到第一个子停下
        while 0 <= r < ROWS and 0 <= c < COLS:
            dest = Point(r, c)
            if board.get(dest) is None:
                moves.append(dest)
            else:
                # 阶段2：找到炮架，继续寻找可吃目标
                r += dr
                c += dc
                while 0 <= r < ROWS and 0 <= c < COLS:
                    dest2 = Point(r, c)
                    piece2 = board.get(dest2)
                    if piece2 is not None:
                        # 关键：隔子吃，且对方子才可吃
                        if piece2.player != player:
                            moves.append(dest2)
                        break
                    r += dr
                    c += dc
                break
            r += dr
            c += dc
    return moves
```

### 2.4 合法性检查方法

#### 两层检查机制

```python
def legal_moves(self):
    """生成合法走法"""
    moves = []
    for point, piece in self.board.pieces_for_player(self.next_player):
        for dest in get_piece_moves(piece, point, self.board):
            move = Move(point, dest)
            # 第一层：规则走法检查（get_piece_moves已完成）
            # 第二层：游戏规则检查
            if self._is_move_legal(move):
                moves.append(move)
    return moves

def _is_move_legal(self, move):
    """检查游戏规则：将帅照面、送将禁止"""
    board = self.board
    captured = board.make_move(move)  # 模拟走子
    legal = True

    # 检查1：将帅照面
    if board.generals_facing():
        legal = False
    # 检查2：己方将是否被吃
    elif board.find_general(self.next_player) is None:
        legal = False
    else:
        # 检查3：对方是否能吃己方将
        gen_pos = board.find_general(self.next_player)
        for pt, pc in board.pieces_for_player(self.next_player.other):
            if gen_pos in get_piece_moves(pc, pt, board):
                legal = False
                break

    board.unmake_move(move, captured)  # 恢复棋盘
    return legal
```

### 2.5 性能优化：make/unmake 模式

```python
def make_move(self, move):
    """原地执行走法，返回被吃的棋子"""
    piece = self._grid.pop(move.from_point)
    captured = self._grid.pop(move.to_point, None)
    self._grid[move.to_point] = piece
    return captured  # 返回被吃子，用于恢复

def unmake_move(self, move, captured):
    """撤销走法"""
    piece = self._grid.pop(move.to_point)
    self._grid[move.from_point] = piece
    if captured is not None:
        self._grid[move.to_point] = captured

# 使用示例
captured = board.make_move(move)
# ... 检查合法性
board.unmake_move(move, captured)  # 零拷贝恢复
```

**优势：** 避免 `deepcopy`，提升搜索性能 10-100 倍。

### 3.2 Alpha-Beta搜索AI (Q2)

```python
class ChessAlphaBetaAgent:
    def select_move(self, game_state: GameState) -> Move:
        # 1. 迭代加深
        for depth in range(1, self.max_depth + 1):
            move, score = self._root_search(game_state, depth)
            if move is not None:
                best_move = move
            # 2. 时间控制
            if self._time_up():
                break
        return best_move

    def _negamax(self, board, player, depth, alpha, beta):
        # 1. 置换表查询
        if key in self._tt:
            return cached_value

        # 2. 终局检测
        if depth == 0:
            return self.evaluate(board, player)

        # 3. 生成并排序走法
        moves = self._sort_moves(legal_moves, board)

        # 4. Negamax + Alpha-Beta剪枝
        for move in moves:
            score = -self._negamax(board, player.other, depth-1, -beta, -alpha)
            alpha = max(alpha, score)
            if alpha >= beta:  # 剪枝
                break

        # 5. 存置换表
        self._tt[key] = (depth, score, flag, best_move)
        return alpha
```

**算法组合：**
- **Negamax**: 简化极大极小算法
- **Alpha-Beta剪枝**: 减少搜索节点
- **迭代加深**: 时间可控
- **置换表**: 避免重复计算
- **走法排序**: MVV-LVA + 历史启发

**性能：** 深度4-5，5000-50000节点/5秒

## 六、设计模式总结

| 模式 | 应用场景 | 代码位置 |
|------|----------|----------|
| **策略模式** | 走法生成（每种棋子独立函数） | `ccmoves.py` |
| **工厂模式** | AI创建（create_agent） | `play_chess.py` |

| **备忘录模式** | 置换表（缓存已计算位置） | `chess_alphabeta_agent.py` |

## 七、性能优化技巧

1. **make/unmake**: 避免deepcopy，性能提升10-100倍
2. **置换表**: 缓存已搜索位置，减少重复计算
3. **走法排序**: MVV-LVA优先吃高价值子，增加剪枝概率
4. **历史启发**: 记录导致剪枝的走法，优先搜索
5. **迭代加深**: 时间可控，逐步增加深度
